SELECT * FROM SOLAR_INFO;

--날짜별 발전량 합계 구하기 
SELECT TO_CHAR(CREATED_AT, 'YYYY-MM-DD'), SUM(SOLAR_WATT) 
FROM SOLAR_INFO
GROUP BY TO_CHAR(CREATED_AT, 'YYYY-MM-DD');

--월별 발전량 합계 구하기
SELECT TO_CHAR(CREATED_AT, 'YYYY-MM'), SUM(SOLAR_WATT) 
FROM SOLAR_INFO
GROUP BY TO_CHAR(CREATED_AT, 'YYYY-MM');

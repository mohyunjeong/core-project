# aws와 라즈베리파이 연동
터미널 창 가서 다음과 같은 명령어 실행
aws configure
access key, secret access key, region name, json 입력
입력후 aws configure 데이터 입력 확인

# 라즈베리파이와 아두이노 우노 연동
터미널 창에서 다음과 같은 명령어 실행
sudo apt-get update
sudo apt-get upgrade
sudo apt-get install arduino

# 아두이노 IDE에서 라이브러리 설치
Arduino IDE 열고
스케치 -> 라이브러리 포함하기 -> 라이브러리 관리 클릭
Servo 검색해서 install
